1 Polar Axis

\mathcal{L} \tag{1}

Code 
import numpy as np
import matplotlib.pyplot as plt

for i in range(2):
    r = np.arange(0, 2, 0.01)
    theta = 8 * np.pi * r
    fig, ax = plt.subplots(
      subplot_kw = {'projection': 'polar'} 
    )
    fig.set_size_inches(2,2)

    ax.plot(theta, r)
    ax.set_rticks([0.5, 1, 1.5, 2])
    ax.grid(True)
    plt.show()
